{{/* album shortcode — links to an album page in the photo library. Usage: {{< album slug="france-2024" >}} {{< album slug="france-2024" title="France 2024" >}} {{< album slug="france-2024" title="France 2024" thumb="true" >}} Parameters: slug (required) URL slug of the album (shown in 'photo album list') title (optional) link text; defaults to "View album" thumb (optional) "true" shows the album cover thumbnail above the link Set site.Params.photoServer in your Hugo config to your photo server base URL: [params] photoServer = "https://photos.yourdomain.com" */}} {{- $slug := .Get "slug" -}} {{- $title := .Get "title" | default "View album" -}} {{- $thumb := .Get "thumb" | default "false" -}} {{- if not $slug -}} {{ errorf "album shortcode: slug parameter is required" }} {{- end -}} {{- $base := "" -}} {{- with site.Params.photoServer -}} {{- $base = . -}} {{- else -}} {{- $base = "https://photos.example.com" -}} {{- end -}} {{- $albumURL := printf "%s/albums/%s" $base $slug -}} {{- if eq $thumb "true" }} {{- /* For thumb we need the cover photo ID — link to album without cover image */}} {{ $title }} {{- else }} {{ $title }} {{- end }}
